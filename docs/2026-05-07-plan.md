import re
import shlex

CHROME_CANDIDATES = [
    "/Applications/Google Chrome.app/Contents/MacOS/Google Chrome",
    "/Applications/Chromium.app/Contents/MacOS/Chromium",
    "/Applications/Microsoft Edge.app/Contents/MacOS/Microsoft Edge",
]

_INVALID_CHARS = re.compile(r'[/\\:*?"<>|\x00-\x1f]')


def sanitize_filename(name: str, max_len: int = 80) -> str:
    """Replace filesystem-invalid characters with `_`, trim to max_len."""
    cleaned = _INVALID_CHARS.sub("_", name).strip()
    return cleaned[:max_len]


def find_chrome() -> str:
    """Return the first existing Chrome-family binary path or exit with a clear message."""
    for path in CHROME_CANDIDATES:
        if Path(path).is_file():
            return path
    sys.exit(
        "找不到 Chrome / Chromium / Edge。请装 Google Chrome:\n"
        "  brew install --cask google-chrome"
    )


def load_config(path: Path) -> dict:
    """Parse a simple KEY=value env file. Supports double-quoted values, '#' comments."""
    out: dict = {}
    if not path.is_file():
        return out
    for raw in path.read_text().splitlines():
        line = raw.strip()
        if not line or line.startswith("#"):
            continue
        if "=" not in line:
            continue
        key, _, value = line.partition("=")
        # shlex.split handles quoted strings cleanly (`"a b"` → `a b`)
        parts = shlex.split(value, posix=True)
        out[key.strip()] = parts[0] if parts else ""
    return out
```

```python
import subprocess
import shutil
import platform


def _run(cmd, *, cwd=None, env=None, check=True, capture=False):
    """Run a shell command list, stream or capture output."""
    print(f"$ {' '.join(map(str, cmd))}")
    if capture:
        return subprocess.run(cmd, cwd=cwd, env=env, check=check, text=True, capture_output=True)
    return subprocess.run(cmd, cwd=cwd, env=env, check=check)


def _which(prog: str):
    """Return path to `prog` or None — typed loosely for Python 3.9 compat."""
    return shutil.which(prog)


def _preflight() -> None:
    if platform.system() != "Darwin":
        sys.exit("仅支持 macOS")
    if not _which("brew"):
        sys.exit("没找到 Homebrew。装一下：https://brew.sh")
    if not _which("node"):
        sys.exit("没找到 Node.js。先装：brew install node")
    out = _run(["node", "--version"], capture=True).stdout.strip()
    major = int(out.lstrip("v").split(".")[0])
    if major < 20:
        sys.exit(f"Node 版本太旧 ({out})，需要 20+。升级：brew upgrade node")
    if not _which("git"):
        sys.exit("没找到 git")
    if not _which("pnpm"):
        print("pnpm 未安装，启用 corepack…")
        _run(["corepack", "enable"])
        _run(["corepack", "prepare", "pnpm@latest", "--activate"])
    find_chrome()  # exits if missing


def _ensure_config_env() -> dict:
    src = REPO_ROOT / "config.example.env"
    dst = REPO_ROOT / "config.env"
    if not dst.exists():
        shutil.copy(src, dst)
        print(f"已生成 config.env（默认值）：{dst}")
    return load_config(dst)


def _install_wewe_rss(cfg: dict) -> None:
    target = REPO_ROOT / "wewe-rss"
    ref = cfg.get("WEWE_RSS_REPO_REF", "v2.6.1")
    if not target.exists():
        _run(["git", "clone", "https://github.com/cooderl/wewe-rss.git", str(target)])
    _run(["git", "fetch", "--tags"], cwd=target)
    _run(["git", "checkout", ref], cwd=target)

    # Switch Prisma schema from MySQL → SQLite
    mysql_dir = target / "apps/server/prisma"
    sqlite_dir = target / "apps/server/prisma-sqlite"
    if sqlite_dir.exists():
        if mysql_dir.exists():
            shutil.rmtree(mysql_dir)
        shutil.move(str(sqlite_dir), str(mysql_dir))

    data_dir = REPO_ROOT / cfg.get("DATA_DIR", "./data").lstrip("./")
    data_dir.mkdir(parents=True, exist_ok=True)

    env_path = target / ".env"
    port = cfg.get("WEWE_RSS_PORT", "4000")
    auth = cfg.get("AUTH_CODE", "mp-pdf-bot")
    db_url = f"file:{data_dir.resolve()}/wewe-rss.db"
    env_path.write_text(
        f"DATABASE_URL={db_url}\n"
        f"DATABASE_TYPE=sqlite\n"
        f"AUTH_CODE={auth}\n"
        f"FEED_MODE=fulltext\n"
        f"SERVER_ORIGIN_URL=http://localhost:{port}\n"
    )

    _run(["pnpm", "install"], cwd=target)
    schema = "apps/server/prisma/schema.prisma"
    _run(["npx", "prisma", "generate", f"--schema={schema}"], cwd=target)
    _run(["npx", "prisma", "migrate", "deploy", f"--schema={schema}"], cwd=target)
    _run(["pnpm", "run", "-r", "build"], cwd=target)


def _install_launchd(cfg: dict) -> None:
    logs_dir = REPO_ROOT / "logs"
    logs_dir.mkdir(exist_ok=True)
    template = (REPO_ROOT / "lib/plist_template.xml").read_text()

    pnpm_path = _which("pnpm") or sys.exit("pnpm 找不到（应该不该到这一步）")
    node_dir = str(Path(_which("node")).parent)
    path_env = f"{node_dir}:/usr/local/bin:/usr/bin:/bin:/opt/homebrew/bin"

    plist = template.format(
        label=LAUNCHD_LABEL,
        pnpm_path=pnpm_path,
        wewe_rss_dir=str(REPO_ROOT / "wewe-rss"),
        path_env=path_env,
        stdout_log=str(logs_dir / "wewe-rss.log"),
        stderr_log=str(logs_dir / "wewe-rss.err.log"),
    )
    PLIST_PATH.parent.mkdir(parents=True, exist_ok=True)
    PLIST_PATH.write_text(plist)

    # bootout if already loaded; ignore failure (not loaded → ok)
    subprocess.run(
        ["launchctl", "bootout", f"gui/{os.getuid()}/{LAUNCHD_LABEL}"],
        capture_output=True,
    )
    _run(["launchctl", "bootstrap", f"gui/{os.getuid()}", str(PLIST_PATH)])


def _install_cron(cfg: dict) -> None:
    cron_time = cfg.get("CRON_TIME", "0 18 * * *")
    sync_log = REPO_ROOT / "logs/sync.log"
    new_line = (
        f'{cron_time} cd {REPO_ROOT} && ./mybot sync >> {sync_log} 2>&1 {CRON_TAG}'
    )

    existing = subprocess.run(
        ["crontab", "-l"], capture_output=True, text=True
    ).stdout
    kept = [ln for ln in existing.splitlines() if CRON_TAG not in ln]
    kept.append(new_line)
    new_crontab = "\n".join(kept) + "\n"
    subprocess.run(["crontab", "-"], input=new_crontab, text=True, check=True)
    print(f"cron 已装：{cron_time}")


def cmd_setup(args):
    print("== 1. 环境检查 ==")
    _preflight()
    print("== 2. 配置 ==")
    cfg = _ensure_config_env()
    print("== 3. 安装 wewe-rss ==")
    _install_wewe_rss(cfg)
    print("== 4. 注册 launchd ==")
    _install_launchd(cfg)
    print("== 5. 装 cron ==")
    _install_cron(cfg)
    print(
        f"\n✅ 装好了。下一步：\n"
        f"  ./mybot open      # 浏览器扫码登录微信读书 + 添加公众号\n"
        f"  ./mybot sync      # 立即拉一次（cron 每天 {cfg.get('CRON_TIME', '0 18 * * *')}）"
    )
```

import sqlite3
import time


def _fetch_articles(db_path: Path):
    con = sqlite3.connect(db_path)
    rows = con.execute("""
        SELECT a.id, a.title, a.publish_time, f.mp_name
          FROM articles a
          JOIN feeds f ON a.mp_id = f.id
         ORDER BY a.publish_time DESC
    """).fetchall()
    con.close()
    return rows


def _render_pdf(chrome: str, url: str, outfile: Path) -> bool:
    cmd = [
        chrome,
        "--headless=new",
        "--disable-gpu",
        "--no-sandbox",
        "--hide-scrollbars",
        "--virtual-time-budget=15000",
        "--run-all-compositor-stages-before-draw",
        "--no-pdf-header-footer",
        f"--print-to-pdf={outfile}",
        url,
    ]
    try:
        subprocess.run(cmd, check=True, capture_output=True, timeout=60)
        return outfile.exists() and outfile.stat().st_size > 1024
    except (subprocess.CalledProcessError, subprocess.TimeoutExpired):
        return False


def cmd_sync(args):
    cfg = load_config(REPO_ROOT / "config.env")
    if not cfg:
        sys.exit("没找到 config.env，先跑 ./mybot setup")

    data_dir = (REPO_ROOT / cfg.get("DATA_DIR", "./data").lstrip("./")).resolve()
    db = data_dir / "wewe-rss.db"
    if not db.exists():
        sys.exit(f"DB 不存在：{db}\n等 wewe-rss 拉过文章再跑 sync")

    out_root = (REPO_ROOT / cfg.get("PDF_OUTPUT_DIR", "./pdfs").lstrip("./")).resolve()
    out_root.mkdir(parents=True, exist_ok=True)
    chrome = find_chrome()

    rows = _fetch_articles(db)
    if not rows:
        print("没有可下载的文章。先到 ./mybot open 加公众号")
        return

    print(f"共 {len(rows)} 篇")
    ok = skip = fail = 0
    for i, (aid, title, ts, mp_name) in enumerate(rows, 1):
        date = time.strftime("%Y%m%d", time.localtime(ts))
        out_dir = out_root / sanitize_filename(mp_name)
        out_dir.mkdir(parents=True, exist_ok=True)
        outfile = out_dir / f"{date}-{sanitize_filename(title)}.pdf"
        url = f"https://mp.weixin.qq.com/s/{aid}"

        if outfile.exists():
            skip += 1
            continue

        print(f"[{i:>3}/{len(rows)}]  →   {outfile.name}", flush=True)
        if _render_pdf(chrome, url, outfile):
            ok += 1
        else:
            fail += 1
            print(f"        ⚠ 失败：{url}")
        time.sleep(0.5)

    print(f"\n完成：成功 {ok} / 跳过 {skip} / 失败 {fail}")
    print(f"输出：{out_root}")
```

    r = subprocess.run(

    )

    db = (REPO_ROOT / cfg.get("DATA_DIR", "./data").lstrip("./") / "wewe-rss.db")
    if db.exists():

    pdfs_dir = (REPO_ROOT / cfg.get("PDF_OUTPUT_DIR", "./pdfs").lstrip("./"))

# mp-pdf-bot

公众号文章自动归档为 PDF。基于 [wewe-rss](https://github.com/cooderl/wewe-rss) + headless Chrome。

## 它做什么

- 你订阅一个微信公众号，wewe-rss 帮你持续拉新文章
- 每天 18:00 自动把新文章渲染为 PDF，按公众号归档到 `./pdfs/`

## 依赖

- macOS（仅）
- Homebrew
- Node.js 20+ — `brew install node`
- Google Chrome — `brew install --cask google-chrome`
- git

## 安装

```bash
git clone <this-repo> mp-pdf-bot
cd mp-pdf-bot
./mybot setup
```

`setup` 会：
- 检查依赖
- 把 wewe-rss 克隆到 `./wewe-rss/` 并构建（1-3 分钟）
- 注册 launchd（开机自启 + 崩了重启）
- 加 cron 每天 18:00 跑一次同步

完成后：
```bash
./mybot open    # 浏览器自动打开 dashboard
```
- 在 dashboard 里：扫码登录微信读书（**不要勾"24 小时自动退出"**）
- 公众号源 → 添加 → 粘贴公众号任意一篇文章的分享链接

## 日常使用

```bash
./mybot sync         # 立即拉一次新文章并转 PDF
./mybot status       # 查看运行状态
./mybot logs         # tail 服务日志
./mybot open         # 打开 dashboard 添加 / 管理公众号
./mybot stop         # 暂停服务
./mybot start        # 启动服务
./mybot uninstall    # 卸载（保留数据和 PDF）
```

PDF 落到 `./pdfs/<公众号名>/<日期>-<标题>.pdf`。

## 配置

`./mybot setup` 会从 `config.example.env` 生成 `config.env`，可改：

```
WEWE_RSS_PORT=4000               # wewe-rss 监听端口
AUTH_CODE=mp-pdf-bot             # 后台访问密码
PDF_OUTPUT_DIR=./pdfs            # PDF 输出目录
DATA_DIR=./data                  # SQLite 数据库目录
WEWE_RSS_REPO_REF=v2.6.1         # wewe-rss 锁定版本
CRON_TIME="0 18 * * *"           # 自动同步时间（标准 cron 表达式）
```

改完重跑 `./mybot setup` 生效。

## 排错

**`./mybot setup` 卡在 pnpm install**：网络问题。配 npm 镜像：
```bash
pnpm config set registry https://registry.npmmirror.com
```

**`./mybot status` 显示 HTTP 502 / no response**：launchd 没起来。
```bash
./mybot logs   # 看 wewe-rss 报什么错
./mybot stop && ./mybot start
```

**PDF 生成全部失败**：mp.weixin.qq.com 文章 key 过期或被风控。等几小时重试，或在 dashboard 里手动同步一次再 sync。

**微信读书账号被风控**：添加公众号太频繁会被封 24 小时。等就好。

## 卸载

```bash
./mybot uninstall                                  # 拆服务，保留数据
rm -rf data/ pdfs/ config.env                      # 完全清理
```
